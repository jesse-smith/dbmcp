**Step B — add the helper.** Add as a private method on `ConnectionManager` (recommended
placement; keeps IO with the rest of the connect layer). Use exactly this structure, adapted
from the research code example:

```python
def _require_databricks_catalog(
    self,
    dialect: "DatabricksDialect",
    *,
    host: str,
    http_path: str,
    token: str,
    schema: str,
    orig_value_error: ValueError,
) -> None:  # NoReturn-style: always raises
    """Raise ConnectionError enriched with accessible-catalog list.

    Called when DatabricksDialect.create_engine raised ValueError because the
    catalog was missing/empty. Builds a bare engine with a placeholder catalog
    ("system" — typically present on Databricks workspaces), runs SHOW CATALOGS
    via dialect.list_catalogs, composes the D-05 message, and raises
    ConnectionError with the original ValueError chained.
    """
    from sqlalchemy.exc import SQLAlchemyError

    hint = "Pass one via ?catalog= in the URL or catalog= in the config."
    probe_engine = None
    try:
        # Placeholder catalog "system" — any connectable Databricks catalog works;
        # "system" is typically present on UC workspaces. If not, the SQLAlchemyError
        # branch below handles it and the user still gets a useful error.
        probe_engine = dialect.create_engine(
            host=host,
            http_path=http_path,
            token=token,
            catalog="system",
            schema=schema or "default",
        )
        catalogs = dialect.list_catalogs(probe_engine)
    except SQLAlchemyError as probe_exc:
        raise ConnectionError(
            f"Databricks connection requires a catalog, and SHOW CATALOGS "
            f"failed ({type(probe_exc).__name__}: {probe_exc}). {hint}"
        ) from orig_value_error
    except Exception as probe_exc:
        raise ConnectionError(
            f"Databricks connection requires a catalog, and probing "
            f"SHOW CATALOGS failed ({type(probe_exc).__name__}: {probe_exc}). "
            f"{hint}"
        ) from orig_value_error
    finally:
        if probe_engine is not None:
            probe_engine.dispose()

    truncated = catalogs[:20]
    suffix = f" (and {len(catalogs) - 20} more)" if len(catalogs) > 20 else ""
    listing = ", ".join(truncated) if truncated else "(none)"
    raise ConnectionError(
        f"Databricks connection requires a catalog. "
        f"Accessible catalogs: {listing}{suffix}. {hint}"
    ) from orig_value_error
```

Exact strings that MUST appear in the message (asserted by tests in Plan 04):
- `"Databricks connection requires a catalog"`
- `"Accessible catalogs:"` (with colon)
- `"SHOW CATALOGS failed"` (in the probe-failure branch)
